import random
import math
import pandas as pd
from typing import List, Dict, Any, Union

def process_rag_row(row: Dict[str, Any]) -> Union[Dict[str, Any], None]:
    """
    Traite une seule ligne du dataset RAG selon les règles de subsampling.
    
    Args:
        row (dict): Dictionnaire contenant 'id', 'question', 'contexts' et 'gold_ids'.
        
    Returns:
        dict | None: La nouvelle ligne traitée ou None si le sample doit être écarté.
    """
    sample_id = row.get("id")
    question = row.get("question")
    contexts = row.get("contexts", [])
    gold_ids = row.get("gold_ids", [])
    
    # Assurer que gold_ids est un ensemble pour des recherches rapides en O(1)
    gold_ids_set = set(gold_ids)
    
    # Séparer les chunks "gold" des chunks "non-gold"
    gold_chunks = []
    non_gold_chunks = []
    
    for chunk in contexts:
        # On gère le cas où l'id de la paire est stocké en chaîne ou en entier
        chunk_id = chunk.get("id")
        if chunk_id in gold_ids_set or str(chunk_id) in gold_ids_set:
            gold_chunks.append(chunk)
        else:
            non_gold_chunks.append(chunk)
            
    n_gold = len(gold_chunks)
    n_non_gold = len(non_gold_chunks)
    
    # --- EDGE CASE 1 : Aucun Gold ID ---
    if n_gold == 0:
        # Prendre un nombre aléatoire de non-gold entre 2 et 10
        # (limité par le nombre de non-golds réellement disponibles)
        n_add = random.randint(2, 10)
        n_add = min(n_non_gold, n_add)
        
        selected_non_gold = random.sample(non_gold_chunks, n_add)
        new_contexts = selected_non_gold
        
    # --- CAS GÉNÉRAL (Avec Gold IDs) ---
    else:
        # Application des équations sur entiers déduites de : 0.2 <= Ratio <= 0.7
        # 1. Pour respecter le max de 70% de golds: n_add >= ceil(3/7 * n_gold)
        min_n_add = math.ceil((3 / 7) * n_gold)
        
        # 2. Pour respecter le min de 20% de golds: n_add <= 4 * n_gold
        # On plafonne aussi par la quantité de non-golds disponibles dans notre enveloppe
        max_n_add = min(n_non_gold, 4 * n_gold)
        
        # --- EDGE CASE 2 : Ratio impossible à tenir (trop de golds, pas assez de non-golds) ---
        if min_n_add > max_n_add:
            # Même en ajoutant le maximum de non-golds possibles, le ratio de golds reste > 70%
            # Nous écartons (discard) ce sample.
            return None
            
        # Sélection aléatoire du nombre de non-golds à ajouter dans l'intervalle valide
        n_add = random.randint(min_n_add, max_n_add)
        
        # Piocher aléatoirement n_add chunks parmi les non_golds
        selected_non_gold = random.sample(non_gold_chunks, n_add)
        
        # Fusionner avec l'intégralité des golds
        new_contexts = gold_chunks + selected_non_gold
        
    # Mélanger les chunks pour que les "golds" ne soient pas toujours au début
    random.shuffle(new_contexts)
    
    # Retourner la structure propre demandée
    return {
        "id": sample_id,
        "question": question,
        "contexts": new_contexts,
        "gold_ids": gold_ids,  # Conserve la liste originale intacte
        "_nb_chunk": len(new_contexts)
    }

def generate_subsampled_dataset(data: Union[List[Dict[str, Any]], pd.DataFrame]) -> pd.DataFrame:
    """
    Prend un dataset (DataFrame ou Liste de dicts), applique les règles de filtrage
    et retourne un DataFrame propre et consolidé.
    """
    # Conversion en liste de dictionnaires si c'est un DataFrame Pandas
    rows = data.to_dict(orient="records") if isinstance(data, pd.DataFrame) else data
    
    processed_rows = []
    discarded_count = 0
    
    for row in rows:
        result = process_rag_row(row)
        if result is not None:
            processed_rows.append(result)
        else:
            discarded_count += 1
            
    print(f"--- Statistiques de traitement ---")
    print(f"Lignes d'origine : {len(rows)}")
    print(f"Lignes conservées : {len(processed_rows)}")
    print(f"Lignes écartées (Ratio gold > 70% inévitable) : {discarded_count}")
    
    if processed_rows:
        avg_chunks = sum(r["_nb_chunk"] for r in processed_rows) / len(processed_rows)
        print(f"Nombre moyen de chunks par échantillon final : {avg_chunks:.2f}")
        
    return pd.DataFrame(processed_rows)

import random
import math
from typing import List, Dict, Any, Union

def process_rag_row(row: Dict[str, Any]) -> Union[Dict[str, Any], None]:
    sample_id = row.get("id")
    question = row.get("question")
    contexts = row.get("contexts", [])
    gold_ids = set(row.get("gold_ids", []))
    
    gold_chunks = []
    non_gold_chunks = []
    
    for chunk in contexts:
        if chunk.get("id") in gold_ids or str(chunk.get("id")) in gold_ids:
            gold_chunks.append(chunk)
        else:
            non_gold_chunks.append(chunk)
            
    n_gold = len(gold_chunks)
    n_non_gold = len(non_gold_chunks)
    
    # Tri par score décroissant
    non_gold_chunks_sorted = sorted(non_gold_chunks, key=lambda x: x.get("score", 0.0), reverse=True)
    
    if n_gold == 0:
        n_add = min(n_non_gold, random.randint(2, 10))
        new_contexts = non_gold_chunks_sorted[:n_add]
    else:
        min_n_add = math.ceil((3 / 7) * n_gold)
        max_n_add = min(n_non_gold, 4 * n_gold)
        
        if min_n_add > max_n_add:
            return None # Discard
            
        n_add = random.randint(min_n_add, max_n_add)
        new_contexts = gold_chunks + non_gold_chunks_sorted[:n_add]
        
    random.shuffle(new_contexts)
    
    return {
        "id": sample_id,
        "question": question,
        "contexts": new_contexts,
        "gold_ids": row.get("gold_ids", []),
        "_nb_chunk": len(new_contexts)
    }



























import random
import math
import pandas as pd

def filtrer_dataset_rag_par_score(df: pd.DataFrame) -> pd.DataFrame:
    """
    Prend le dataset RAG (DataFrame) en entrée, applique la logique de ratio
    et de sélection par le score, et retourne le nouveau DataFrame.
    """
    new_rows = []
    
    # On boucle sur chaque ligne du dataset (DataFrame)
    for index, row in df.iterrows():
        sample_id = row.get("id")
        question = row.get("question")
        contexts = row.get("contexts", [])
        gold_ids = row.get("gold_ids", [])
        
        gold_ids_set = set(gold_ids)
        
        # 1. Séparer les golds et les non-golds
        gold_chunks = []
        non_gold_chunks = []
        
        for chunk in contexts:
            chunk_id = chunk.get("id")
            if chunk_id in gold_ids_set or str(chunk_id) in gold_ids_set:
                gold_chunks.append(chunk)
            else:
                non_gold_chunks.append(chunk)
                
        n_gold = len(gold_chunks)
        n_non_gold = len(non_gold_chunks)
        
        # 2. Trier les non-golds par score décroissant (les meilleurs en premier)
        non_gold_chunks_sorted = sorted(non_gold_chunks, key=lambda x: x.get("score", 0.0), reverse=True)
        
        # 3. Appliquer les règles de sélection
        if n_gold == 0:
            # Si pas de gold, on prend entre 2 et 10 non-golds parmi les meilleurs
            n_add = min(n_non_gold, random.randint(2, 10))
            new_contexts = non_gold_chunks_sorted[:n_add]
        else:
            # Calcul des bornes pour respecter les 20% à 70% de golds
            min_n_add = math.ceil((3 / 7) * n_gold)
            max_n_add = min(n_non_gold, 4 * n_gold)
            
            # EDGE CASE : Si c'est mathématiquement impossible d'avoir moins de 70% de golds
            if min_n_add > max_n_add:
                continue # On discard le sample (on passe à la ligne suivante sans l'ajouter)
                
            n_add = random.randint(min_n_add, max_n_add)
            # On garde tous les golds + les Top-K non-golds
            new_contexts = gold_chunks + non_gold_chunks_sorted[:n_add]
            
        # 4. Mélanger les paires/triplets pour casser le biais de position
        random.shuffle(new_contexts)
        
        # 5. On prépare la nouvelle ligne
        new_rows.append({
            "id": sample_id,
            "question": question,
            "contexts": new_contexts,
            "gold_ids": gold_ids, # Reste inchangé
            "_nb_chunk": len(new_contexts) # Nouvelle colonne demandée
        })
        
    # On reconstruit le nouveau DataFrame à partir des lignes conservées
    return pd.DataFrame(new_rows)
