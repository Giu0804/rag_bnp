import numpy as np
import torch

def run_seek_v2_for_nli(df, w=1, k=1, p_main=97.8, p_backoff=90.0):
    results = []
    
    for idx, row in df.iterrows():
        claims = parse_atomic_claims(row['atomic_claims'])
        chunks_list = row['contexts']
        
        if len(claims) == 0 or len(chunks_list) == 0:
            results.append({**row.to_dict(), "chunks_filtered": []})
            continue
            
        # 1. Extraction et mapping des unités de chunks
        all_sentences = []
        sentence_chunk_ids = []
        
        for chunk in chunks_list:
            sentences = split_insurance_text(chunk['text'])
            all_sentences.extend(sentences)
            sentence_chunk_ids.extend([chunk['id']] * len(sentences))
            
        if len(all_sentences) == 0:
            results.append({**row.to_dict(), "chunks_filtered": []})
            continue
            
        # 2. Embeddings et similarité brute (Calcul SEEK standard)
        e_claims = embed_model.encode(claims, convert_to_tensor=True)
        e_sentences = embed_model.encode(all_sentences, convert_to_tensor=True)
        
        # Fenêtre w=1
        window_embeddings = []
        N = len(all_sentences)
        for i in range(N):
            start = max(0, i - w)
            end = min(N, i + w + 1)
            window_embeddings.append(torch.mean(e_sentences[start:end], dim=0))
        window_embeddings = torch.stack(window_embeddings)
        
        # Matrice de similarité (M x N) en Float32
        window_embeddings_norm = torch.nn.functional.normalize(window_embeddings, p=2, dim=1)
        e_claims_norm = torch.nn.functional.normalize(e_claims, p=2, dim=1)
        similarity_matrix = torch.mm(e_claims_norm.float(), window_embeddings_norm.T.float()).cpu().numpy()
        
        # 3. Double filtrage asymétrique (Main vs Backoff)
        sentences_to_keep_per_chunk = {c['id']: set() for c in chunks_list}
        
        for claim_idx in range(len(claims)):
            scores_for_claim = similarity_matrix[claim_idx]
            
            tau_main = np.percentile(scores_for_claim, p_main)
            tau_backoff = np.percentile(scores_for_claim, p_backoff)
            
            for i in range(N):
                c_id = sentence_chunk_ids[i]
                score = scores_for_claim[i]
                
                if score >= tau_main or score >= tau_backoff:
                    sentences_to_keep_per_chunk[c_id].add(i)

        # 4. Recomposition du texte filtré avec ajout des voisins (contrôlés par w)
        chunks_filtered = []
        for chunk in chunks_list:
            c_id = chunk['id']
            
            # On récupère toutes les phrases de ce chunk spécifique
            chunk_indices = [i for i, chunk_id in enumerate(sentence_chunk_ids) if chunk_id == c_id]
            chunk_indices_set = set(chunk_indices)
            
            # On ne garde que celles validées par les percentiles
            validated_indices = sentences_to_keep_per_chunk[c_id].intersection(chunk_indices_set)
            
            # --- AJOUT DES VOISINS (SÉCURISÉ PAR CHUNK & SANS DOUBLONS) ---
            extended_indices = set()
            for idx_val in validated_indices:
                # Génération de la fenêtre [idx_val - w, ..., idx_val + w]
                # max et min évitent de sortir de l'index global de all_sentences
                start_w = max(0, idx_val - w)
                end_w = min(N, idx_val + w + 1)
                
                for neighbor_idx in range(start_w, end_w):
                    # Condition CRITIQUE : Le voisin doit appartenir AU MÊME chunk
                    if neighbor_idx in chunk_indices_set:
                        extended_indices.add(neighbor_idx)
            
            # Le tri (sorted) garantit l'ordre chronologique des phrases dans le texte final
            valid_indices = sorted(list(extended_indices))
            
            if valid_indices:
                filtered_text = " ".join([all_sentences[i] for i in valid_indices])
                chunks_filtered.append({
                    "id": c_id,
                    "text": filtered_text
                })
                
        results.append({
            "question": row['question'],
            "contexts": chunks_list,
            "generated_answer": row['generated_answer'],
            "atomic_claims": row['atomic_claims'],
            "GT_cited": row.get('GT_cited', []),
            "chunks_filtered": chunks_filtered
        })
        
    return pd.DataFrame(results)

def run_seek_attribution(df, w=3, k=3, percentile_p=95):
    results = []
    
    for idx, row in df.iterrows():
        claims = parse_atomic_claims(row['atomic_claims'])
        chunks_list = row['contexts']
        
        if not claims or not chunks_list:

                "contexts": chunks_list,
                "generated_answer": row['generated_answer'],
                "atomic_claims": row.get('atomic_claims', ''),
                "GT_cited": row.get('GT_cited', []),
                "ablation_scores": {},
                "pred_cited": [],
                "chunks_filtered": [] # Ajouté pour la cohérence
            })
            continue
            
        # Étape A : Séquence globale de phrases et mapping des IDs
        all_sentences = []
        sentence_chunk_ids = []
        
        for chunk in chunks_list:
            c_id = chunk['id']
            sentences = split_insurance_text(chunk['text'])
            all_sentences.extend(sentences)
            sentence_chunk_ids.extend([c_id] * len(sentences))
            
        if not all_sentences:
            results.append({
                "question": row['question'],
                "contexts": chunks_list,
                "generated_answer": row['generated_answer'],
                "atomic_claims": row.get('atomic_claims', ''),
                "GT_cited": row.get('GT_cited', []),
                "ablation_scores": {},
                "pred_cited": [],
                "chunks_filtered": [] # Ajouté pour la cohérence
            })
            continue
            
        # Étape B : Vectorisation Qwen 3
        e_claims = embed_model.encode(claims, prompt_name="query", convert_to_tensor=True)
        e_sentences = embed_model.encode(all_sentences, convert_to_tensor=True)
        
        # Étape C : Fenêtres contextuelles glissantes SEEK
        N = len(all_sentences)
        window_embeddings = []
        for i in range(N):
            start = max(0, i - w)
            end = min(N, i + w + 1)
            v_i = torch.mean(e_sentences[start:end], dim=0)
            window_embeddings.append(v_i)
        window_embeddings = torch.stack(window_embeddings)
        
        # Étape D : Similarité brute (Cosinus)
        window_embeddings_norm = torch.nn.functional.normalize(window_embeddings, p=2, dim=1)
        e_claims_norm = torch.nn.functional.normalize(e_claims, p=2, dim=1)
        similarity_matrix = torch.mm(e_claims_norm, window_embeddings_norm.T).cpu().numpy()
        
        # Étape E : Lissage Temporel (Smoothing)
        smoothed_matrix = np.zeros_like(similarity_matrix)
        half_k = k // 2
        for i in range(N):
            start = max(0, i - half_k)
            end = min(N, i + half_k + 1)
            smoothed_matrix[:, i] = np.mean(similarity_matrix[:, start:end], axis=1)
            
        # Étape F : Application du Seuil et Max-Pooling par Chunk ID
        chunk_scores = {chunk['id']: 0.0 for chunk in chunks_list}
        pred_cited_set = set()
        
        # Structure pour mémoriser les indices de phrases validées par chunk
        sentences_to_keep_per_chunk = {chunk['id']: set() for chunk in chunks_list}
        
        for claim_idx in range(len(claims)):
            scores_for_claim = smoothed_matrix[claim_idx]
            tau = np.percentile(scores_for_claim, percentile_p)
            
            for i in range(N):
                if scores_for_claim[i] >= tau:
                    c_id = sentence_chunk_ids[i]
                    pred_cited_set.add(c_id) 
                    sentences_to_keep_per_chunk[c_id].add(i) # Stockage de la phrase validée
                    
                    if scores_for_claim[i] > chunk_scores[c_id]:
                        chunk_scores[c_id] = float(np.round(scores_for_claim[i], 4))
                        
        # Étape F.bis : Recomposition du texte filtré avec voisins internes au chunk
        chunks_filtered = []
        for chunk in chunks_list:
            c_id = chunk['id']
            
            # Liste et set des indices de ce chunk spécifique dans all_sentences
            chunk_indices = [i for i, chunk_id in enumerate(sentence_chunk_ids) if chunk_id == c_id]
            chunk_indices_set = set(chunk_indices)
            
            # Intersection pour avoir les indices validés pour ce chunk
            validated_indices = sentences_to_keep_per_chunk[c_id].intersection(chunk_indices_set)
            
            extended_indices = set()
            for idx_val in validated_indices:
                # Fenêtre autour de la phrase validée (on utilise le paramètre w de la fonction)
                start_w = max(0, idx_val - w)
                end_w = min(N, idx_val + w + 1)
                
                for neighbor_idx in range(start_w, end_w):
                    # SÉCURITÉ : Reste strictement dans le même chunk
                    if neighbor_idx in chunk_indices_set:
                        extended_indices.add(neighbor_idx)
                        
            valid_indices = sorted(list(extended_indices))
            
            if valid_indices:
                filtered_text = " ".join([all_sentences[i] for i in valid_indices])
                chunks_filtered.append({
                    "id": c_id,
                    "text": filtered_text
                })
                                
        # Étape G : Stockage complet avec colonnes d'évaluation
        results.append({
            "question": row['question'],
            "contexts": chunks_list,
            "generated_answer": row['generated_answer'],
            "atomic_claims": row['atomic_claims'],
            "GT_cited": row.get('GT_cited', []),       
            "ablation_scores": chunk_scores,           
            "pred_cited": list(pred_cited_set),        
            "chunks_filtered": chunks_filtered         # Intégré ici !
        })
        
    return pd.DataFrame(results)
