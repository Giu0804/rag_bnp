import re
import numpy as np
import pandas as pd
import torch
from sentence_transformers import SentenceTransformer

# 1. Chargement du modèle d'embedding Qwen 3
embed_model = SentenceTransformer(
    "Qwen/Qwen3-Embedding-0.6B",
    model_kwargs={"attn_implementation": "flash_attention_2", "device_map": "auto"},
    tokenizer_kwargs={"padding_side": "left"},
)

# 2. Découpage adapté aux listes et tableaux d'assurance
def split_insurance_text(text):
    pattern = r'(?<=\.)\s+|(?<=;)\s+|\n\s*(?=[-•*])'
    return [el.strip() for el in re.split(pattern, text) if el.strip()]

# 3. Parsing des faits atomiques
def parse_atomic_claims(claims_text):
    if not isinstance(claims_text, str): return []
    return [line.lstrip('- ').strip() for line in claims_text.split('\n') if line.strip()]

# 4. Pipeline SEEK Inversé (Centré sur l'Évidence)
def run_seek_attribution(df, w=3, k=3, percentile_p=95):
    results = []
    
    for idx, row in df.iterrows():
        claims = parse_atomic_claims(row['atomic_claims'])
        chunks_list = row['contexts']
        
        if not claims or not chunks_list:
            results.append({
                "question": row['question'],
                "contexts": chunks_list,
                "generated_answer": row['generated_answer'],
                "atomic_claims": row.get('atomic_claims', ''),
                "GT_cited": row.get('GT_cited', []),
                "ablation_scores": {},
                "pred_cited": []
            })
            continue
            
        # Étape A : Séquence globale de phrases et mapping des IDs
        all_sentences = []
        sentence_chunk_ids = []
        
        for chunk in chunks_list:
            c_id = chunk['id']
            sentences = split_insurance_text(chunk['text'])
            all_sentences.extend(sentences)
            sentence_chunk_ids.extend([c_id] * len(sentences))
            
        if not all_sentences:
            results.append({
                "question": row['question'],
                "contexts": chunks_list,
                "generated_answer": row['generated_answer'],
                "atomic_claims": row.get('atomic_claims', ''),
                "GT_cited": row.get('GT_cited', []),
                "ablation_scores": {},
                "pred_cited": []
            })
            continue
            
        # Étape B : Vectorisation Qwen 3
        e_claims = embed_model.encode(claims, prompt_name="query", convert_to_tensor=True)
        e_sentences = embed_model.encode(all_sentences, convert_to_tensor=True)
        
        # Étape C : Fenêtres contextuelles glissantes SEEK
        N = len(all_sentences)
        window_embeddings = []
        for i in range(N):
            start = max(0, i - w)
            end = min(N, i + w + 1)
            v_i = torch.mean(e_sentences[start:end], dim=0)
            window_embeddings.append(v_i)
        window_embeddings = torch.stack(window_embeddings)
        
        # Étape D : Similarité brute (Cosinus)
        window_embeddings_norm = torch.nn.functional.normalize(window_embeddings, p=2, dim=1)
        e_claims_norm = torch.nn.functional.normalize(e_claims, p=2, dim=1)
        similarity_matrix = torch.mm(e_claims_norm, window_embeddings_norm.T).cpu().numpy()
        
        # Étape E : Lissage Temporel (Smoothing)
        smoothed_matrix = np.zeros_like(similarity_matrix)
        half_k = k // 2
        for i in range(N):
            start = max(0, i - half_k)
            end = min(N, i + half_k + 1)
            smoothed_matrix[:, i] = np.mean(similarity_matrix[:, start:end], axis=1)
            
        # Étape F : Application du Seuil et Max-Pooling par Chunk ID
        chunk_scores = {chunk['id']: 0.0 for chunk in chunks_list}
        pred_cited_set = set()
        
        for claim_idx in range(len(claims)):
            scores_for_claim = smoothed_matrix[claim_idx]
            tau = np.percentile(scores_for_claim, percentile_p)
            
            for i in range(N):
                # Si la paire fait-unité dépasse le seuil adaptatif du percentile
                if scores_for_claim[i] >= tau:
                    c_id = sentence_chunk_ids[i]
                    pred_cited_set.add(c_id) # Noté en citation
                    
                    # On garde le score lissé maximum obtenu toutes paires confondues
                    if scores_for_claim[i] > chunk_scores[c_id]:
                        chunk_scores[c_id] = float(np.round(scores_for_claim[i], 4))
                        
        # Étape G : Stockage complet avec colonnes d'évaluation
        results.append({
            "question": row['question'],
            "contexts": chunks_list,
            "generated_answer": row['generated_answer'],
            "atomic_claims": row['atomic_claims'],
            "GT_cited": row.get('GT_cited', []),       # Ton Ground Truth initial
            "ablation_scores": chunk_scores,           # Dictionnaire {id: score_max}
            "pred_cited": list(pred_cited_set)         # Liste des IDs qui ont dépassé le seuil
        })
        
    return pd.DataFrame(results)



















def compute_metrics(df_res):
    precisions, recalls, f1_scores, exact_matches = [], [], [] , []
    
    def clean_to_set(val):
        if val is None or (isinstance(val, float) and np.isnan(val)):
            return set()
        return set(val)
    
    for _, row in df_res.iterrows():
        gt = clean_to_set(row.get('GT_cited', []))
        pred = clean_to_set(row.get('pred_cited', []))
        
        # 1. Calcul de l'Accuracy (Exact Match)
        em = 1.0 if gt == pred else 0.0
        exact_matches.append(em)
        
        # 2. Calcul des autres métriques
        if not pred:
            precision = 1.0 if not gt else 0.0
            recall = 1.0 if not gt else 0.0
        else:
            intersection = gt.intersection(pred)
            precision = len(intersection) / len(pred)
            recall = len(intersection) / len(gt) if gt else 1.0
            
        f1 = (2 * precision * recall) / (precision + recall) if (precision + recall) > 0 else 0.0
        
        precisions.append(precision)
        recalls.append(recall)
        f1_scores.append(f1)
        
    print(f"Précision Moyenne : {np.mean(precisions):.2%}")
    print(f"Rappel Moyen      : {np.mean(recalls):.2%}")
    print(f"F1-Score Moyen    : {np.mean(f1_scores):.2%}")
    print(f"Accuracy (EM)     : {np.mean(exact_matches):.2%}")









def compute_metrics_grid(df_res):
    precisions, recalls, f1_scores, exact_matches = [], [], [], []
    
    def clean_to_set(val):
        if val is None or (isinstance(val, float) and np.isnan(val)):
            return set()
        return set(val)
    
    for _, row in df_res.iterrows():
        gt = clean_to_set(row.get('GT_cited', []))
        pred = clean_to_set(row.get('pred_cited', []))
        
        em = 1.0 if gt == pred else 0.0
        exact_matches.append(em)
        
        if not pred:
            precision = 1.0 if not gt else 0.0
            recall = 1.0 if not gt else 0.0
        else:
            intersection = gt.intersection(pred)
            precision = len(intersection) / len(pred)
            recall = len(intersection) / len(gt) if gt else 1.0
            
        f1 = (2 * precision * recall) / (precision + recall) if (precision + recall) > 0 else 0.0
        
        precisions.append(precision)
        recalls.append(recall)
        f1_scores.append(f1)
        
    # On renvoie les moyennes pour le suivi du Grid Search
    return np.mean(precisions), np.mean(recalls), np.mean(f1_scores), np.mean(exact_matches)
