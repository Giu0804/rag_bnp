import numpy as np
import pandas as pd
import torch
from FlagEmbedding import BGEM3FlagModel

# Initialisation de BGE-M3
embed_model = BGEM3FlagModel("BAAI/bge-m3", use_fp16=True)


def run_seek_attribution_bge_m3(df, w=1, k=3, percentile_p=95):
    results = []

    for idx, row in df.iterrows():
        claims = parse_atomic_claims(row["atomic_claims"])
        chunks_list = row["contexts"]

        if not claims or not chunks_list:
            results.append({
                "question": row["question"],
                "contexts": chunks_list,
                "generated_answer": row["generated_answer"],
                "atomic_claims": row.get("atomic_claims", ""),
                "GT_cited": row.get("GT_cited", []),
                "ablation_scores": {},
                "pred_cited": [],
                "filtered_chunks": [],  # Format [{"id": int, "text": str}]
            })
            continue

        # Étape A : Séquence globale de phrases et mapping des IDs
        all_sentences = []
        sentence_chunk_ids = []

        for chunk in chunks_list:
            c_id = chunk["id"]
            sentences = split_insurance_text(chunk["text"])
            all_sentences.extend(sentences)
            sentence_chunk_ids.extend([c_id] * len(sentences))

        N = len(all_sentences)
        if N == 0:
            results.append({
                "question": row["question"],
                "contexts": chunks_list,
                "generated_answer": row["generated_answer"],
                "atomic_claims": row.get("atomic_claims", ""),
                "GT_cited": row.get("GT_cited", []),
                "ablation_scores": {},
                "pred_cited": [],
                "filtered_chunks": [],
            })
            continue

        # Étape B & C : Fenêtres contextuelles intégrées (texte continu i-w à i+w)
        integrated_windows = []
        for i in range(N):
            start = max(0, i - w)
            end = min(N, i + w + 1)
            window_text = " ".join(all_sentences[start:end])
            integrated_windows.append(window_text)

        # Vectorisation ColBERT (Multi-Vector) avec BGE-M3
        out_claims = embed_model.encode(
            claims, return_dense=False, return_colbert_vecs=True
        )
        out_windows = embed_model.encode(
            integrated_windows, return_dense=False, return_colbert_vecs=True
        )

        # Étape D : Matrice de Similarité Brute via ColBERT MaxSim
        similarity_matrix = np.zeros((len(claims), N))
        for claim_idx, q_vec in enumerate(out_claims["colbert_vecs"]):
            for win_idx, d_vec in enumerate(out_windows["colbert_vecs"]):
                score = embed_model.colbert_score(q_vec, d_vec).item()
                similarity_matrix[claim_idx, win_idx] = score

        # Étape E : Lissage Temporel (Smoothing k=3)
        smoothed_matrix = np.zeros_like(similarity_matrix)
        half_k = k // 2
        for i in range(N):
            start = max(0, i - half_k)
            end = min(N, i + half_k + 1)
            smoothed_matrix[:, i] = np.mean(
                similarity_matrix[:, start:end], axis=1
            )

        # Étape F : Seuil, Max-Pooling et Extraction des Sous-Portions Validées
        chunk_scores = {chunk["id"]: 0.0 for chunk in chunks_list}
        pred_cited_set = set()

        # Dictionnaire pour collecter les phrases retenues par chunk ID : {c_id: {index_phrase: texte_phrase}}
        valid_sentences_per_chunk = {chunk["id"]: {} for chunk in chunks_list}

        for claim_idx in range(len(claims)):
            scores_for_claim = smoothed_matrix[claim_idx]
            tau = np.percentile(scores_for_claim, percentile_p)

            for i in range(N):
                # Si la paire fait-fenêtre dépasse le seuil adaptatif du percentile
                if scores_for_claim[i] >= tau:
                    c_id = sentence_chunk_ids[i]
                    pred_cited_set.add(c_id)

                    # Stockage de la phrase originale responsable du dépassement de seuil
                    # (On utilise l'index 'i' pour conserver l'ordre d'apparition original)
                    valid_sentences_per_chunk[c_id][i] = all_sentences[i]

                    # Mise à jour du score max du chunk
                    if scores_for_claim[i] > chunk_scores[c_id]:
                        chunk_scores[c_id] = float(
                            np.round(scores_for_claim[i], 4)
                        )

        # Construction de la liste 'filtered_chunks' au format exact [{"id": c_id, "text": ...}]
        filtered_chunks_list = []
        for chunk in chunks_list:
            c_id = chunk["id"]
            if c_id in pred_cited_set:
                # Récupération des phrases validées triées par leur ordre d'index original 'i'
                sentence_dict = valid_sentences_per_chunk[c_id]
                sorted_indices = sorted(sentence_dict.keys())

                # Concaténation des sous-portions dans l'ordre naturel d'apparition
                concatenated_text = " ".join([
                    sentence_dict[idx] for idx in sorted_indices
                ])

                filtered_chunks_list.append({
                    "id": c_id,
                    "text": concatenated_text,
                })

        # Étape G : Stockage complet
        results.append({
            "question": row["question"],
            "contexts": chunks_list,
            "generated_answer": row["generated_answer"],
            "atomic_claims": row["atomic_claims"],
            "GT_cited": row.get("GT_cited", []),
            "ablation_scores": chunk_scores,
            "pred_cited": list(pred_cited_set),
            "filtered_chunks": filtered_chunks_list,  # Liste nettoyée et ordonnée
        })

    return pd.DataFrame(results)

# VERSION 1 : Moyenne des Embeddings dans la Fenêtre (Moyenne Vectorielle)
# =====================================================================
def run_seek_ot_averaged(
    df, embed_model, w=1, k=3, mass_threshold=0.01, percentile_fallback=90
):
    """Pipeline SEEK avec Optimal Transport.

    Version Vectorielle : Calcule l'embedding de chaque phrase puis applique
    torch.mean sur la fenêtre [i-w : i+w+1].
    """
    results = []

    for idx, row in df.iterrows():
        claims = parse_atomic_claims(row["atomic_claims"])
        chunks_list = row["contexts"]

        if not claims or not chunks_list:
            results.append({
                "question": row["question"],
                "contexts": chunks_list,
                "generated_answer": row["generated_answer"],
                "atomic_claims": row.get("atomic_claims", ""),
                "GT_cited": row.get("GT_cited", []),
                "ablation_scores": {},
                "pred_cited": [],
                "filtered_chunks": [],
                "pi_matrix": None,
            })
            continue

        # Étape A : Séquence globale de phrases et mapping des IDs
        all_sentences = []
        sentence_chunk_ids = []

        for chunk in chunks_list:
            c_id = chunk["id"]
            sentences = split_insurance_text(chunk["text"])
            all_sentences.extend(sentences)
            sentence_chunk_ids.extend([c_id] * len(sentences))

        N = len(all_sentences)
        if N == 0:
            results.append({
                "question": row["question"],
                "contexts": chunks_list,
                "generated_answer": row["generated_answer"],
                "atomic_claims": row.get("atomic_claims", ""),
                "GT_cited": row.get("GT_cited", []),
                "ablation_scores": {},
                "pred_cited": [],
                "filtered_chunks": [],
                "pi_matrix": None,
            })
            continue

        # Étape B : Vectorisation individuelle des phrases
        e_claims = embed_model.encode(
            claims, prompt_name="query", convert_to_tensor=True
        )
        e_sentences = embed_model.encode(
            all_sentences, convert_to_tensor=True
        )

        # Étape C : Fenêtres contextuelles par MOYENNE des embeddings
        window_embeddings = []
        for i in range(N):
            start = max(0, i - w)
            end = min(N, i + w + 1)
            v_i = torch.mean(e_sentences[start:end], dim=0)
            window_embeddings.append(v_i)
        window_embeddings = torch.stack(window_embeddings)

        # Étape D : Similarité brute (Cosinus)
        window_embeddings_norm = torch.nn.functional.normalize(
            window_embeddings, p=2, dim=1
        )
        e_claims_norm = torch.nn.functional.normalize(e_claims, p=2, dim=1)
        similarity_matrix = (
            torch.mm(e_claims_norm, window_embeddings_norm.T).cpu().numpy()
        )

        # Étape E : Lissage Temporel (Smoothing k=3)
        smoothed_matrix = np.zeros_like(similarity_matrix)
        half_k = k // 2
        for i in range(N):
            start = max(0, i - half_k)
            end = min(N, i + half_k + 1)
            smoothed_matrix[:, i] = np.mean(
                similarity_matrix[:, start:end], axis=1
            )

        # -----------------------------------------------------------------
        # Étape F : TRANSPORT OPTIMAL (Earth Mover's Distance)
        # -----------------------------------------------------------------
        num_claims = len(claims)
        num_windows = N

        # Matrice de coût (1 - similarity), normalisée [0, 1] pour la stabilité
        norm_smoothed = (smoothed_matrix - smoothed_matrix.min()) / (
            smoothed_matrix.max() - smoothed_matrix.min() + 1e-8
        )
        cost_matrix = 1.0 - norm_smoothed

        # Distributions de masse uniforme (somme = 1)
        a = np.ones(num_claims) / num_claims
        b = np.ones(num_windows) / num_windows

        # Résolution de l'OT
        pi_matrix = ot.emd(a, b, cost_matrix)

        # Extraction des citations et construction des filtered_chunks
        chunk_scores = {chunk["id"]: 0.0 for chunk in chunks_list}
        pred_cited_set = set()
        valid_sentences_per_chunk = {chunk["id"]: {} for chunk in chunks_list}

        for claim_idx in range(num_claims):
            transport_masses = pi_matrix[claim_idx]

            for win_idx in range(num_windows):
                if transport_masses[win_idx] > mass_threshold:
                    c_id = sentence_chunk_ids[win_idx]
                    pred_cited_set.add(c_id)
                    valid_sentences_per_chunk[c_id][win_idx] = all_sentences[
                        win_idx
                    ]

                    score_val = smoothed_matrix[claim_idx, win_idx]
                    if score_val > chunk_scores[c_id]:
                        chunk_scores[c_id] = float(np.round(score_val, 4))

        # Reconstitution de filtered_chunks au format [{"id": c_id, "text": texte}]
        filtered_chunks_list = []
        for chunk in chunks_list:
            c_id = chunk["id"]
            if c_id in pred_cited_set:
                sentence_dict = valid_sentences_per_chunk[c_id]
                sorted_indices = sorted(sentence_dict.keys())
                concatenated_text = " ".join([
                    sentence_dict[idx] for idx in sorted_indices
                ])
                filtered_chunks_list.append({
                    "id": c_id,
                    "text": concatenated_text,
                })

        # Étape G : Stockage
        results.append({
            "question": row["question"],
            "contexts": chunks_list,
            "generated_answer": row["generated_answer"],
            "atomic_claims": row["atomic_claims"],
            "GT_cited": row.get("GT_cited", []),
            "ablation_scores": chunk_scores,
            "pred_cited": list(pred_cited_set),
            "filtered_chunks": filtered_chunks_list,
            "pi_matrix": pi_matrix,
        })

    return pd.DataFrame(results)

def run_seek_ot_integrated(
    df, embed_model, w=1, k=3, mass_threshold=0.01, percentile_fallback=90
):

    Version Contexte Intégré : Concatène physiquement le texte des sous-portions
    adjacentes (i-w à i+w) avant d'encoder le bloc complet d'un coup.

        norm_smoothed = (smoothed_matrix - smoothed_matrix.min()) / (

        )
        cost_matrix = 1.0 - norm_smoothed

        a = np.ones(num_claims) / num_claims

        pi_matrix = ot.emd(a, b, cost_matrix)

        # Extraction des citations et construction des filtered_chunks

        for claim_idx in range(num_claims):

import matplotlib.pyplot as plt
import numpy as np
import seaborn as sns


def plot_optimal_transport_sample(df_results, row_index=0, max_claim_len=40):
    """Génère une heatmap montrant l'alignement entre faits atomiques et sous-

    portions de chunks via la matrice de transport pi_matrix.
    """
    sample = df_results.iloc[row_index]
    pi_matrix = sample.get("pi_matrix", None)

    if pi_matrix is None:
        print(f"Pas de pi_matrix disponible pour l'index {row_index}.")
        return

    # 1. Extraction et formatage des faits atomiques
    claims = parse_atomic_claims(sample["atomic_claims"])
    claims_labels = [
        f"Fait {i+1}: "
        + (
            c[:max_claim_len] + "..."
            if len(c) > max_claim_len
            else c
        )
        for i, c in enumerate(claims)
    ]

    # 2. Construction des labels pour les sous-portions (Fenêtres)
    num_windows = pi_matrix.shape[1]
    window_labels = [f"Sous-portion {j+1}" for j in range(num_windows)]

    # 3. Création du graphique
    fig, ax = plt.subplots(
        figsize=(max(8, num_windows * 0.8), max(5, len(claims) * 0.8))
    )

    # Heatmap de la matrice pi
    sns.heatmap(
        pi_matrix,
        annot=True,
        fmt=".3f",
        cmap="Blues",
        cbar_kws={"label": "Masse de transport (Alignement)"},
        yticklabels=claims_labels,
        xticklabels=window_labels,
        linewidths=0.5,
        ax=ax,
    )

    # Métadonnées sur le graphique
    gt_cited = sample.get("GT_cited", [])
    pred_cited = sample.get("pred_cited", [])

    plt.title(
        f"Plan de Transport Optimal (Sample {row_index})\n"
        f"GT Cited: {gt_cited} | Pred Cited: {pred_cited}",
        fontsize=12,
        pad=15,
    )
    plt.xlabel("Sous-portions de Chunks (Fenêtres)", fontsize=10)
    plt.ylabel("Faits Atomiques", fontsize=10)
    plt.xticks(rotation=45, ha="right")
    plt.yticks(rotation=0)

    plt.tight_layout()
    plt.show()






import matplotlib.pyplot as plt
import numpy as np
import seaborn as sns


def plot_optimal_transport_sample(
    df_results, row_index=0, aggregate_by_chunk=True, max_claim_len=35
):
    """Visualise le Plan de Transport Optimal (pi_matrix).

    Si aggregate_by_chunk=True, somme la masse de transport par Chunk ID pour
    garder une heatmap compacte et lisible.
    """
    sample = df_results.iloc[row_index]
    pi_matrix = sample.get("pi_matrix", None)

    if pi_matrix is None:
        print(f"Pas de pi_matrix disponible pour l'index {row_index}.")
        return

    claims = parse_atomic_claims(sample["atomic_claims"])
    chunks_list = sample["contexts"]

    # Reconstitution du mapping phrase -> chunk_id
    all_sentences = []
    sentence_chunk_ids = []
    for chunk in chunks_list:
        c_id = chunk["id"]
        sentences = split_insurance_text(chunk["text"])
        all_sentences.extend(sentences)
        sentence_chunk_ids.extend([c_id] * len(sentences))

    # Formater les labels de l'axe Y (Faits atomiques)
    claims_labels = [
        f"Fait {i+1}: "
        + (
            c[:max_claim_len] + "..."
            if len(c) > max_claim_len
            else c
        )
        for i, c in enumerate(claims)
    ]

    # --- CAS 1 : Agrégation par Chunk (Heatmap Compacte) ---
    if aggregate_by_chunk:
        unique_chunk_ids = [chunk["id"] for chunk in chunks_list]
        num_claims = len(claims)
        num_chunks = len(unique_chunk_ids)

        # Création de la matrice agrégée (nb_claims x nb_chunks)
        aggregated_pi = np.zeros((num_claims, num_chunks))

        for c_idx, c_id in enumerate(unique_chunk_ids):
            # Trouver les indices des phrases appartenant à ce chunk
            window_indices = [
                idx
                for idx, cid in enumerate(sentence_chunk_ids)
                if cid == c_id
            ]
            if window_indices:
                # Somme de la masse OT envoyée à toutes les sous-portions de ce chunk
                aggregated_pi[:, c_idx] = np.sum(
                    pi_matrix[:, window_indices], axis=1
                )

        plot_matrix = aggregated_pi
        x_labels = [f"Chunk {c_id}" for c_id in unique_chunk_ids]
        title_suffix = "(Agrégé par Chunk)"

    # --- CAS 2 : Niveau fin (Toutes les sous-portions) ---
    else:
        plot_matrix = pi_matrix
        x_labels = [
            f"S{j+1} (C{sentence_chunk_ids[j]})"
            for j in range(pi_matrix.shape[1])
        ]
        title_suffix = "(Détail Sous-portions)"

    # Ajustement dynamique de la taille de l'image
    fig_width = max(6, len(x_labels) * 1.2)
    fig_height = max(4, len(claims) * 0.8)

    fig, ax = plt.subplots(figsize=(fig_width, fig_height))

    # Dessin de la Heatmap
    sns.heatmap(
        plot_matrix,
        annot=True,
        fmt=".3f",
        cmap="YlGnBu",
        cbar_kws={"label": "Masse de transport"},
        yticklabels=claims_labels,
        xticklabels=x_labels,
        linewidths=0.8,
        ax=ax,
    )

    gt_cited = sample.get("GT_cited", [])
    pred_cited = sample.get("pred_cited", [])

    plt.title(
        f"Plan de Transport Optimal {title_suffix} - Sample {row_index}\n"
        f"GT Cited: {gt_cited} | Pred Cited: {pred_cited}",
        fontsize=11,
        pad=12,
    )
    plt.xlabel("Documents Sources (Chunks)", fontsize=10)
    plt.ylabel("Faits Atomiques", fontsize=10)
    plt.xticks(rotation=0)
    plt.yticks(rotation=0)

    plt.tight_layout()
    plt.show()
